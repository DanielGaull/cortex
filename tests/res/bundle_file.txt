bundle Box {
    value: number,

    fn increment(amount: number) {
        this.value += amount;
    }
}

bundle BoxedBox {
    box: &mut Box,

    fn getBox(): &mut Box {
        this.box
    }
}

fn getBox(value: number): &mut Box {
    Box {
        value: value,
    }
}
fn incValue(box: &mut Box, amt: number): void {
    box.value += amt;
}

// Increments v by 10
fn runTest(v: number): number {
    let box = getBox(v);
    incValue(box, 1); // v + 1
    box.increment(3); // v + 4
    let doubleBox = BoxedBox {box:box};
    doubleBox.getBox().increment(6); // v + 10

    box.value
}
