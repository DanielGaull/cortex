bundle Box {
    value: number,

    fn increment(&mut this, amount: number) {
        this.value += amount;
    }
}

bundle BoxedBox {
    box: &mut Box,

    fn getBox(&mut this): &mut Box {
        this.box
    }
}

bundle Access {
    dummy: number,

    fn __index(&this, val: number): number {
        val
    }
}

fn getBox(value: number): &mut Box {
    Box {
        value: value,
    }
}
fn incValue(box: &mut Box, amt: number): void {
    box.value += amt;
}
