bundle Box {
    value: number,

    fn increment(amount: number) {
        this.value += amount;
    }
}

fn getBox(value: number): Box {
    Box {
        value: value,
    }
}
fn incValue(box: Box, amt: number): void {
    box.value += amt;
}

fn runTest(v: number): number {
    let box = getBox(v);
    incValue(box, 7);
    box.increment(3);
    box.value
}
