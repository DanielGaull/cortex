import mylib;
import drawing::Point as DrawPoint;

// Should just return the value that came into it
fn doBox(v: i32): i32 {
    let box = getBox(v);
    box.get()
}

fn doBox2(v: i32): i32 {
    let box: &mylib::LibBox<i32> = getBox(v);
    readBox(box)
}

fn numWrapper(v: i32): i32 {
    let a: follows AFunc = heap NumWrapper { value: 5 };
    a.a(v)
}

fn hello(): string {

}

// Again, returns what was passed into it in a roundabout way
fn doPoint(x: i32, y: i32): (i32, i32) {
    let p = heap DrawPoint {
        coords: (x, y),
    };
    (p.getX(), p.getY())
}
