module GeometryExtended {
    module Geometry {
        struct Point {
            x: i32,
            y: i32,

            fn incX(&mut this, amount: i32) {
                this.x += amount;
            }
        }
    }
}

extend GeometryExtended::Geometry::Point {
    fn incY(&mut this, amount: i32) {
        this.y += amount;
    }
}

fn main(): &GeometryExtended::Geometry::Point {
    let point = heap GeometryExtended::Geometry::Point {
        x: 1,
        y: 1,
    };
    point.incX(3);
    point.incY(3);
    
    point
}
