fn identity<T>(item: T): T {
    item
}

struct Box<T> {
    item: T,

    fn get(&this): T {
        this.item
    }
    fn set(&mut this, value: T) {
        this.item = value;
    }

    fn identity<R>(&this, value: R): R {
        value
    }
}

fn readBox<T>(box: &Box<T>): T {
    box.get()
}

fn getFirst<T, R>(value: (T, R)): T {
    value.t0
}
